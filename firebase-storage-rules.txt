rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    
    // Client files: Only the client can read/write their own files
    // Admins (authors) can also access
    match /clients/{userId}/{allPaths=**} {
      allow read: if request.auth != null && 
                     (request.auth.uid == userId || 
                      firestore.get(/databases/(default)/documents/authors/$(request.auth.uid)).data != null);
      allow write: if request.auth != null && 
                      (request.auth.uid == userId || 
                       firestore.get(/databases/(default)/documents/authors/$(request.auth.uid)).data != null);
    }
    
    // Blog images: Anyone can read, only admins can write
    match /blogImages/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null && 
                      firestore.get(/databases/(default)/documents/authors/$(request.auth.uid)).data != null;
    }
    
    // Default: Deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
